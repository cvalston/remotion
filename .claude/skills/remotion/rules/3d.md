## Installation

```bash
npx remotion add @remotion/three # npm
bunx remotion add @remotion/three # bun
yarn remotion add @remotion/three # yarn
pnpm exec remotion add @remotion/three # pnpm
```

## Key Requirements

**ThreeCanvas Setup**: Wrap all 3D content in `<ThreeCanvas>` with required `width` and `height` props:

```tsx
import { ThreeCanvas } from "@remotion/three";
import { useVideoConfig } from "remotion";

const { width, height } = useVideoConfig();

<ThreeCanvas width={width} height={height}>
  <ambientLight intensity={0.5} />
  <pointLight position={[10, 10, 10]} />
  <MyMesh />
</ThreeCanvas>
```

**Animation Driver**: Shaders, models, etc. MUST NOT animate by themselves or use `useFrame()`. Drive all animations through `useCurrentFrame()` to prevent flickering during rendering:

```tsx
import { useCurrentFrame } from "remotion";
import { useThree } from "@react-three/fiber";

const MyMesh = () => {
  const frame = useCurrentFrame();
  const rotationY = frame * 0.02;

  return (
    <mesh rotation={[0, rotationY, 0]}>
      <boxGeometry args={[1, 1, 1]} />
      <meshStandardMaterial color="orange" />
    </mesh>
  );
};
```

**Sequence Layout**: When using `<Sequence>` inside `<ThreeCanvas>`, set the `layout` prop to `"none"`:

```tsx
<Sequence layout="none" from={30}>
  <MyMesh />
</Sequence>
```

## Full Example

```tsx
import { AbsoluteFill, useCurrentFrame, useVideoConfig } from "remotion";
import { ThreeCanvas } from "@remotion/three";

export const ThreeScene = () => {
  const { width, height } = useVideoConfig();
  const frame = useCurrentFrame();

  return (
    <AbsoluteFill>
      <ThreeCanvas width={width} height={height}>
        <ambientLight intensity={0.5} />
        <directionalLight position={[5, 5, 5]} />
        <RotatingBox frame={frame} />
      </ThreeCanvas>
    </AbsoluteFill>
  );
};

const RotatingBox = ({ frame }: { frame: number }) => {
  return (
    <mesh rotation={[frame * 0.01, frame * 0.02, 0]}>
      <boxGeometry />
      <meshStandardMaterial color="#4a90e2" />
    </mesh>
  );
};
```
